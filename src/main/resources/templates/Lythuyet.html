thẻ th:src ="@{/path}" dùng để thay đổi thuộc tính của thẻ th:href = "@{/path}" tác dụng tương tự
thẻ th:attr="id = ${id động}, class=${class động}"
thẻ th:with="tênBiến = ${giá trị động}"
th:switch = ${condition} th:case=" "
